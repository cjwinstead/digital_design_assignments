    \item Hierarchical Design
    \begin{lstlisting}
    `timescale 1ns / 1ps

    module carry_select(
        input [3:0] a,
        input [3:0] b,
        input cin,
        output [3:0] sum,
        output cout
    );
    
        wire cout1;
        wire cout2;
        wire [3:0] sum1;
        wire [3:0] sum2;
        
        ripple_carry U1(.a(a), .b(b), .cin(0), .sum(sum1), .cout(cout1));
        ripple_carry U2(.a(a), .b(b), .cin(1), .sum(sum2), .cout(cout2));

        mux MUX0(.a(sum1[0]), .b(sum2[0]), .out(sum[0]));
        mux MUX1(.a(sum1[1]), .b(sum2[1]), .out(sum[1]));
        mux MUX2(.a(sum1[2]), .b(sum2[2]), .out(sum[2]));
        mux MUX3(.a(sum1[3]), .b(sum2[3]), .out(sum[3]));

        mux MUXC(.a(cout1), .b(cout2), .out(cout));

    endmodule
\end{lstlisting}
